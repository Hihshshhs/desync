    local c=lp.Character; local h=c and c:FindFirstChild("HumanoidRootPart")
    if not State.desync_active and h then
        State.desync_active=true; State.frozenPos=h.Position
        State.distanceCounter=0; State.lastSyncTime=tick(); State.lastPulse=tick()
    end
    if State.desync_active and h then
        local d=(h.Position-State.frozenPos).Magnitude
        local t=tick()-State.lastSyncTime
        if d>=(State.distanceCounter+Config.SyncInterval) or t>=Config.TimeInterval then
            forceSync(); State.frozenPos=h.Position; State.distanceCounter=0; State.lastSyncTime=tick()
        end
    end
    if State.desync_active then ghostify() end
    if Config.AnimsDisabled then
        local hum=c and c:FindFirstChildOfClass("Humanoid")
        local anim=hum and hum:FindFirstChildOfClass("Animator")
        if anim then for _,t in pairs(anim:GetPlayingAnimationTracks()) do pcall(function() t:Stop(0) end) end end
    end
    -- Update status
    local on=State.desync_active
    statusLbl.Text=on and "DESYNC: ON  |  Ghost active" or "DESYNC: OFF"
    statusLbl.TextColor3=on and Theme.Accent1 or Theme.TextSecondary
    statusDot.BackgroundColor3=on and Theme.Success or Theme.Error
    indLbl.Visible=on
    -- pulse indicator border
    if on then
        local p=(math.sin(tick()*3)+1)/2
        indStroke.Color=Color3.fromHSV(0.5+p*0.08,1,1)
    end
end))

-- ── Heartbeat ─────────────────────────────────────────────────────────────────
track(RunService.Heartbeat:Connect(function()
    local c=lp.Character; local h=c and c:FindFirstChild("HumanoidRootPart")
    if not State.desync_active and h then
        State.desync_active=true; State.frozenPos=h.Position
        State.distanceCounter=0; State.lastSyncTime=tick(); State.lastPulse=tick()
    end
    if State.desync_active and h then
        local d=(h.Position-State.frozenPos).Magnitude
        local t=tick()-State.lastSyncTime
        if d>=(State.distanceCounter+Config.SyncInterval) or t>=Config.TimeInterval then
            forceSync(); State.frozenPos=h.Position; State.distanceCounter=0; State.lastSyncTime=tick()
        end
    end
    if State.desync_active then ghostify() end
    if Config.AnimsDisabled then
        local hum=c and c:FindFirstChildOfClass("Humanoid")
        local anim=hum and hum:FindFirstChildOfClass("Animator")
        if anim then for _,t in pairs(anim:GetPlayingAnimationTracks()) do pcall(function() t:Stop(0) end) end end
    end
    -- Update status
    local on=State.desync_active
    statusLbl.Text=on and "DESYNC: ON  |  Ghost active" or "DESYNC: OFF"
    statusLbl.TextColor3=on and Theme.Accent1 or Theme.TextSecondary
    statusDot.BackgroundColor3=on and Theme.Success or Theme.Error
    indLbl.Visible=on
    -- pulse indicator border
    if on then
        local p=(math.sin(tick()*3)+1)/2
        indStroke.Color=Color3.fromHSV(0.5+p*0.08,1,1)
    end
end))

        local g=CoreGui:FindFirstChild(n); if g then g:Destroy() end
    end
    _G.KryoCleanup=nil
end
