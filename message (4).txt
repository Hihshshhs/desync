local function createButton(name, text, color, pos, parent)
    local btn = Instance.new("TextButton")
    btn.Name = name
    btn.Size = UDim2.new(1, 0, 0, 40)
    btn.Position = pos
    btn.BackgroundColor3 = color
    btn.BorderSizePixel = 0
    btn.Text = text
    btn.TextColor3 = Color3.fromRGB(255, 255, 255)
    btn.Font = Enum.Font.GothamBold
    btn.TextSize = 14
    btn.AutoButtonColor = false
    btn.Parent = parent
    
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 8)
    corner.Parent = btn
    
    -- Touch feedback
    btn.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
            TweenService:Create(btn, TweenInfo.new(0.1), {
                Size = UDim2.new(1, -4, 0, 38),
                Position = UDim2.new(0, 2, pos.Y.Scale, pos.Y.Offset + 1)
            }):Play()
        end
    end)
    
    btn.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
            TweenService:Create(btn, TweenInfo.new(0.1), {
                Size = UDim2.new(1, 0, 0, 40),
                Position = pos
            }):Play()
        end
    end)
    
    return btn
end

local function updateUI()
    if DESYNC_ENABLED then
        statusLabel.Text = "ENABLED"
        statusLabel.TextColor3 = Color3.fromRGB(80, 255, 80)
        statusDot.BackgroundColor3 = Color3.fromRGB(80, 255, 80)
        toggleBtn.Text = "⏹ DISABLE DESYNC"
        toggleBtn.BackgroundColor3 = Color3.fromRGB(60, 180, 60)
    else
        statusLabel.Text = "DISABLED"
        statusLabel.TextColor3 = Color3.fromRGB(255, 80, 80)
        statusDot.BackgroundColor3 = Color3.fromRGB(255, 60, 60)
        toggleBtn.Text = "▶ ENABLE DESYNC"
        toggleBtn.BackgroundColor3 = Color3.fromRGB(220, 60, 60)
    end
    
    if MARKER_ENABLED then
        markerBtn.Text = "👁 MARKER: ON"
        markerBtn.BackgroundColor3 = Color3.fromRGB(80, 80, 110)
    else
        markerBtn.Text = "👁 MARKER: OFF"
        markerBtn.BackgroundColor3 = Color3.fromRGB(50, 50, 65)
    end
    
    -- Animate status dot
    TweenService:Create(statusDot, TweenInfo.new(0.2), {
        BackgroundColor3 = DESYNC_ENABLED and Color3.fromRGB(80, 255, 80) or Color3.fromRGB(255, 60, 60)
    }):Play()
end

local function toggleDesync()
    DESYNC_ENABLED = not DESYNC_ENABLED
    
    if DESYNC_ENABLED then
        serverPosition = hrp.Position
        realPosition = hrp.Position
    end
    
    updateUI()
end

local function rollbackToServerPos()
    if character and hrp then
        -- Flash effect
        TweenService:Create(rollbackBtn, TweenInfo.new(0.1), {BackgroundColor3 = Color3.fromRGB(100, 180, 255)}):Play()
        wait(0.1)
        TweenService:Create(rollbackBtn, TweenInfo.new(0.1), {BackgroundColor3 = Color3.fromRGB(50, 110, 240)}):Play()
        
        hrp.CFrame = CFrame.new(serverPosition)
        realPosition = serverPosition
        print("[DESYNC] Rolled back to server position")
    end
end

local function toggleMarker()
    MARKER_ENABLED = not MARKER_ENABLED
    marker.Transparency = MARKER_ENABLED and 0.5 or 1
    beam.Enabled = MARKER_ENABLED
    billboard.Enabled = MARKER_ENABLED
    updateUI()
end

local function resetServerPos()
    if hrp then
        serverPosition = hrp.Position
        realPosition = hrp.Position
        
        -- Flash effect
        TweenService:Create(resetBtn, TweenInfo.new(0.1), {BackgroundColor3 = Color3.fromRGB(180, 120, 240)}):Play()
        wait(0.1)
        TweenService:Create(resetBtn, TweenInfo.new(0.1), {BackgroundColor3 = Color3.fromRGB(120, 70, 180)}):Play()
        
        print("[DESYNC] Server position reset to current position")
    end
end

local function minimizeUI()
    uiMinimized = true
    mainFrame.Visible = false
    floatBtn.Visible = true
end

local function expandUI()
    uiMinimized = false
    mainFrame.Visible = true
    floatBtn.Visible = false
end

-- ========================
-- MAIN LOOP
-- ========================
RunService.Heartbeat:Connect(function()
    if not character or not hrp then return end
    if not hrp:IsDescendantOf(workspace) then return end
    
    local humanoid = character:FindFirstChild("Humanoid")
    
    -- Always update marker visuals
    marker.CFrame = CFrame.new(serverPosition)
    
    -- Update labels
    local dist = (hrp.Position - serverPosition).Magnitude
    distLabel.Text = string.format("📍 Distance: %.1f studs", dist)
    serverLabel.Text = string.format("🎯 Server: %d, %d, %d", 
        math.floor(serverPosition.X), 
        math.floor(serverPosition.Y), 
        math.floor(serverPosition.Z))
    
    if DESYNC_ENABLED then
        realPosition = hrp.Position
        local distance = (realPosition - serverPosition).Magnitude
        
        if distance >= UPDATE_DISTANCE then
            if FREEZE_AT_HALFWAY and destination then
                local halfway = (serverPosition + realPosition) / 2
                serverPosition = halfway
            else
                serverPosition = realPosition
            end
        end
        
        -- Desync trick
        spawn(function()
            hrp:SetNetworkOwner(nil)
            local frozenCFrame = hrp.CFrame
            hrp.CFrame = CFrame.new(serverPosition)
            wait(0.01)
            hrp.CFrame = frozenCFrame
            hrp:SetNetworkOwner(player)
        end)
    end
    
    if humanoid and humanoid.MoveDirection.Magnitude > 0 then
        destination = hrp.Position + (humanoid.MoveDirection * 100)
    end
end)
