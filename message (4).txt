local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local TweenService = game:GetService("TweenService")
local UserInputService = game:GetService("UserInputService")

local player = Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local hrp = character:WaitForChild("HumanoidRootPart")

-- CONFIG
local DESYNC_ENABLED = false
local UPDATE_DISTANCE = 50
local FREEZE_AT_HALFWAY = true

-- VARIABLES
local realPosition = hrp.Position
local fakePosition = hrp.Position
local destination = nil
local isMoving = false

-- MARKER (server‑visible)
local marker = Instance.new("Part")
marker.Name = "DesyncMarker"
marker.Shape = Enum.PartShape.Ball
marker.Size = Vector3.new(4, 4, 4)
marker.BrickColor = BrickColor.new("Bright red")
marker.Material = Enum.Material.Neon
marker.Anchored = true
marker.CanCollide = false
marker.Transparency = 0.4
marker.Parent = workspace

-- UI (Mobile‑first ScreenGui)
local screenGui = Instance.new("ScreenGui")
screenGui.Name = "DesyncMobileUI"
screenGui.ResetOnSpawn = false
screenGui.Parent = player:WaitForChild("PlayerGui")

-- Main frame (mobile sized)
local frame = Instance.new("Frame")
frame.Size = UDim2.new(0, 240, 0, 160)
frame.Position = UDim2.new(0.5, -120, 0.85, 0) -- bottom‑center on mobile
frame.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
frame.BorderSizePixel = 0
frame.Parent = screenGui

-- Title
local title = Instance.new("TextLabel")
title.Size = UDim2.new(1, 0, 0, 28)
title.BackgroundTransparency = 1
title.Text = "DESYNC CONTROLS"
title.TextColor3 = Color3.fromRGB(255, 255, 255)
title.Font = Enum.Font.GothamBold
title.TextSize = 16
title.Parent = frame

-- Status label
local statusLabel = Instance.new("TextLabel")
statusLabel.Size = UDim2.new(1, 0, 0, 28)
statusLabel.Position = UDim2.new(0, 0, 0, 35)
statusLabel.BackgroundTransparency = 1
statusLabel.Text = "Status: DISABLED"
statusLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
statusLabel.Font = Enum.Font.Gotham
statusLabel.TextSize = 14
statusLabel.TextXAlignment = Enum.TextXAlignment.Left
statusLabel.Parent = frame

-- Toggle button (touch area)
local toggleBtn = Instance.new("TextButton")
toggleBtn.Size = UDim2.new(1, 0, 0, 36)
toggleBtn.Position = UDim2.new(0, 0, 0, 70)
toggleBtn.BackgroundColor3 = Color3.fromRGB(100, 100, 100)
toggleBtn.Text = "TOGGLE DESYNC"
toggleBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
toggleBtn.Font = Enum.Font.GothamBold
toggleBtn.TextSize = 15
toggleBtn.Parent = frame

-- Rollback button (touch area)
local rollbackBtn = Instance.new("TextButton")
rollbackBtn.Size = UDim2.new(1, 0, 0, 36)
rollbackBtn.Position = UDim2.new(0, 0, 0, 112)
rollbackBtn.BackgroundColor3 = Color3.fromRGB(50, 150, 50)
rollbackBtn.Text = "ROLLBACK TO SERVER POS"
rollbackBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
rollbackBtn.Font = Enum.Font.GothamBold
rollbackBtn.TextSize = 14
rollbackBtn.Parent = frame

-- Helper: update marker (always at fake pos)
local function updateMarker()
    marker.CFrame = CFrame.new(fakePosition)
end

-- Helper: set fake position
local function setFakePosition(pos)
    fakePosition = pos
    hrp:SetNetworkOwner(nil)
    hrp.CFrame = CFrame.new(fakePosition)
    hrp:SetNetworkOwner(player)
    updateMarker()
end

-- Helper: update status UI
local function updateStatus(enabled)
    DESYNC_ENABLED = enabled
    statusLabel.Text = "Status: " .. (enabled and "ENABLED" or "DISABLED")
    statusLabel.TextColor3 = enabled and Color3.fromRGB(100, 255, 100) or Color3.fromRGB(255, 100, 100)
    toggleBtn.Text = enabled and "DISABLE DESYNC" or "ENABLE DESYNC"
    toggleBtn.BackgroundColor3 = enabled and Color3.fromRGB(150, 50, 50) or Color3.fromRGB(100, 100, 100)
end

-- Update desync loop
local function updateDesync()
    if not DESYNC_ENABLED then return end
    
    local currentPos = hrp.Position
    local distance = (currentPos - realPosition).Magnitude
    
    if distance >= UPDATE_DISTANCE then
        if FREEZE_AT_HALFWAY and destination then
            local halfway = (realPosition + destination) / 2
            setFakePosition(halfway)
        else
            setFakePosition(currentPos)
        end
        realPosition = currentPos
    end
end

-- MAIN LOOP
RunService.Heartbeat:Connect(function()
    if not character or not hrp then return end
    
    local humanoid = character:FindFirstChild("Humanoid")
    if humanoid and humanoid.MoveDirection.Magnitude > 0 then
        isMoving = true
        destination = hrp.Position + (humanoid.MoveDirection * 100)
        updateDesync()
    else
        isMoving = false
    end
end)

-- MOBILE TOUCH HANDLING (no keybinds)
toggleBtn.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.Touch then
        updateStatus(not DESYNC_ENABLED)
    end
end)

rollbackBtn.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.Touch then
        if not DESYNC_ENABLED then
            statusLabel.Text = "Rollback: No desync active"
            wait(2)
            statusLabel.Text = "Status: DISABLED"
            return
        end
        
        -- Force fake position to match real position (server sees you)
        setFakePosition(realPosition)
        updateStatus(false)
        statusLabel.Text = "Rollback: Position synced to server"
        wait(2)
        statusLabel.Text = "Status: DISABLED"
    end
end)

-- INITIALIZE
updateStatus(DESYNC_ENABLED)
marker.Parent = workspace
screenGui.Parent = player:WaitForChild("PlayerGui")

print("[DESYNC + MOBILE UI] Loaded — Marker at workspace.DesyncMarker | Touch buttons to control (no keybinds)")
