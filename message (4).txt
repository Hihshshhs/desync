--[[
    Delta Optimizer - FPS Boost & Script Load Accelerator
    For Delta Executor on Roblox
]]

-- Priority loader - runs immediately
local Optimizer = {
    Name = "DeltaOptimizer",
    Version = "2.0",
    LoadTime = tick()
}

-- Core optimization functions
local function ApplyGraphicsOptimizations()
    -- Disable shadows for FPS boost
    if game:GetService("Lighting") then
        game.Lighting.GlobalShadows = false
        game.Lighting.FogEnd = 1000
        game.Lighting.Brightness = 1.5
    end
    
    -- Reduce render distance for low-end devices
    if game:GetService("Workspace") then
        workspace.CurrentCamera.CameraType = Enum.CameraType.Custom
        workspace.DescendantAdded:Connect(function(desc)
            if desc:IsA("Part") and not desc:IsA("BasePart") then
                desc.Material = Enum.Material.SmoothPlastic
            end
        end)
    end
end

local function RemoveUnnecessaryEffects()
    -- Remove decals and particles
    for _, v in pairs(game:GetDescendants()) do
        if v:IsA("Decal") or v:IsA("Texture") then
            v:Destroy()
        elseif v:IsA("ParticleEmitter") or v:IsA("Fire") or v:IsA("Smoke") then
            v.Enabled = false
        elseif v:IsA("MeshPart") then
            v.Material = Enum.Material.SmoothPlastic
        elseif v:IsA("Clothing") then
            v:Destroy()
        end
    end
end

local function MemoryOptimization()
    -- Clear unused instances
    game:GetService("RunService").Heartbeat:Connect(function()
        -- Auto cleanup every 5 seconds
        if tick() % 5 < 0.1 then
            for _, v in pairs(game:GetDescendants()) do
                if v:IsA("Sound") and v.Playing == false and v.TimePosition == 0 then
                    v:Destroy()
                end
            end
        end
    end)
end

local function AccelerateScriptLoading()
    -- Increase script priority
    local runService = game:GetService("RunService")
    local replicatedStorage = game:GetService("ReplicatedStorage")
    
    -- Fast script injection point
    local scriptContainer = Instance.new("Folder")
    scriptContainer.Name = "DeltaOptimizedScripts"
    scriptContainer.Parent = replicatedStorage
    
    -- Override require function for faster loading
    local originalRequire = require
    getgenv().require = function(module)
        if module and module.Parent then
            return originalRequire(module)
        end
        return nil
    end
    
    -- Priority executor
    local fastSpawn = function(func)
        if runService:IsRunning() then
            spawn(func)
        else
            task.spawn(func)
        end
    end
    
    getgenv().FastSpawn = fastSpawn
end

local function DisableUnnecessaryServices()
    local services = {
        "Chat",
        "Players",
        "UserInputService"
    }
    
    -- Keep essential services, disable heavy ones
    pcall(function()
        local players = game:GetService("Players")
        players.CharacterAutoLoads = false
    end)
    
    -- Reduce render quality further if needed
    local userSettings = UserSettings()
    if userSettings then
        local gameSettings = userSettings:GetService("UserGameSettings")
        if gameSettings then
            gameSettings.MaterialQuality = Enum.MaterialQuality.Low
            gameSettings.VisualQualityLevel = 1
        end
    end
end

local function NetworkOptimization()
    -- Reduce network updates for FPS
    local function modifyNetworkSettings()
        local players = game:GetService("Players")
        for _, player in pairs(players:GetPlayers()) do
            if player ~= players.LocalPlayer then
                local character = player.Character
                if character then
                    for _, part in pairs(character:GetDescendants()) do
                        if part:IsA("BasePart") then
                            part.NetworkOwnershipRule = Enum.NetworkOwnershipRule.Manual
                        end
                    end
                end
            end
        end
    end
    
    modifyNetworkSettings()
    
    players.PlayerAdded:Connect(modifyNetworkSettings)
    players.PlayerCharacterAdded:Connect(modifyNetworkSettings)
end

-- Main execution - runs before other scripts
local function InitializeOptimizer()
    -- Set priority flags
    getgenv().DeltaOptimizerLoaded = true
    getgenv().OptimizationLevel = "Maximum"
    
    -- Apply all optimizations
    ApplyGraphicsOptimizations()
    RemoveUnnecessaryEffects()
    MemoryOptimization()
    AccelerateScriptLoading()
    DisableUnnecessaryServices()
    NetworkOptimization()
    
    -- Monitor and maintain FPS
    local frames = 0
    local lastTime = tick()
    
    game:GetService("RunService").RenderStepped:Connect(function()
        frames = frames + 1
        if tick() - lastTime >= 1 then
            getgenv().CurrentFPS = frames
            frames = 0
            lastTime = tick()
            
            -- Dynamic optimization based on FPS
            if getgenv().CurrentFPS < 30 then
                -- Additional aggressive optimizations
                pcall(function()
                    workspace.GlobalWind = nil
                    workspace.Gravity = workspace.Gravity
                end)
            end
        end
    end)
    
    print(string.format("[Delta Optimizer] Loaded in %.3f seconds", tick() - Optimizer.LoadTime))
    print("[Delta Optimizer] Optimizations active - FPS boost enabled")
end

-- Execute optimizer immediately with highest priority
if not getgenv().DeltaOptimizerLoaded then
    InitializeOptimizer()
    
    -- Hook to ensure optimizer runs before other scripts
    local oldPCall = pcall
    getgenv().pcall = function(func, ...)
        if getgenv().DeltaOptimizerLoaded then
            return oldPCall(func, ...)
        end
        return false, "Optimizer not loaded"
    end
    
    -- Force priority execution
    task.wait(0) -- Minimal yield to ensure execution
    InitializeOptimizer() -- Double run for safety
else
    print("[Delta Optimizer] Already running")
end

-- Export functions for other scripts
getgenv().GetOptimizerStats = function()
    return {
        LoadTime = Optimizer.LoadTime,
        Version = Optimizer.Version,
        CurrentFPS = getgenv().CurrentFPS or 0,
        OptimizationActive = true
    }
end

return getgenv().GetOptimizerStats
